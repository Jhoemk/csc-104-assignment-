// Function to calculate the area of a rectangle function calculateArea(length, width) { return length * width; } // Function to print the result function printResult(area) { console.log("The area of the rectangle is: " + area); } // Main function function main() { var length = 5; // Length of the rectangle var width = 3; // Width of the rectangle var area = calculateArea(length, width); // Calculate the area printResult(area); // Print the result } // Call the main function main();
